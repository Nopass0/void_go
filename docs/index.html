VoidDB Go SDK

Official Go companion client for VoidDB with document CRUD, fluent queries, schema metadata, cache access, Blob field uploads, and `vdbgo` commands for schema pull and Go model generation.

GitHub Core server Server docs

Install

go get github.com/Nopass0/void_go
go install github.com/Nopass0/void_go/cmd/vdbgo@latest

Quick start

client, err := voidorm.New(voidorm.Config{
    URL: "http://localhost:7700",
})
if err != nil {
    panic(err)
}

_, err = client.Login(context.Background(), "admin", "admin")
if err != nil {
    panic(err)
}

users := client.DB("app").Collection("users")

id, err := users.Insert(context.Background(), voidorm.Doc{
    "name": "Alice",
    "active": true,
})

rows, err := users.Query(
    context.Background(),
    voidorm.NewQuery().
        Where("active", voidorm.Eq, true).
        OrderBy("name", voidorm.Asc).
        Limit(25),
)

Environment-first setup

VOIDDB_URL=https://db.lowkey.su
VOIDDB_USERNAME=admin
VOIDDB_PASSWORD=your-password
client, err := voidorm.NewFromEnv()
if err != nil {
    panic(err)
}

_, err = client.LoginFromEnv(context.Background())

Blob fields and uploads

ref, err := client.DB("media").Collection("assets").UploadFile(
    context.Background(),
    "asset-123",
    "original",
    strings.NewReader("hello voiddb"),
    voidorm.UploadFileOptions{
        Filename: "hello.txt",
        ContentType: "text/plain",
    },
)
if err != nil {
    panic(err)
}

fmt.Println(ref.URL)

Query JSON

q := voidorm.NewQuery().
    Where("age", voidorm.Gte, 18).
    Limit(10)

payload, err := q.JSON()
